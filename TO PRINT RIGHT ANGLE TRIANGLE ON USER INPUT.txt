import java.util.Scanner;
public class GT {
    public static void main(String args[])
    {
        int k;
        Scanner sc = new Scanner(System.in);
		System.out.print(“Enter the number:”);
		k = sc.nextInt(); 
        int a, b;
        for (a = 0; a < k; a++) {
            for (b = 0; b <= a; b++) {
                System.out.print("* ");
            }
            System.out.println();
        }
    }
}
